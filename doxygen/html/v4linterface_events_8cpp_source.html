360  void V4LInterface::__set_control(std::list<ControlHolder> * list, Gtk::Widget * wctrl) {
361  std::list<ControlHolder>::iterator iter;
362 
363  for (iter = list->begin(); iter != list->end(); ++iter) {
364  if ((*iter).widget == wctrl) break;
365  }
366 
367  int value;
368  struct v4l2_queryctrl qctrl = (*iter).qctrl;
369 
370  switch (qctrl.type) {
371  case V4L2_CTRL_TYPE_INTEGER:
372  case V4L2_CTRL_TYPE_INTEGER64:
373  case V4L2_CTRL_TYPE_CTRL_CLASS:
374  case V4L2_CTRL_TYPE_BITMASK:
375  default:
376  break;
377 
378  case V4L2_CTRL_TYPE_BOOLEAN:
379 
380  value = static_cast<Gtk::CheckButton *>(wctrl)->get_active();
381  if (!vcap.set_control(qctrl.id, value))
382  std::cout << "Can not update control [" << qctrl.name << "] with value " << value << std::endl;
383 
384  break;
385 
386  case V4L2_CTRL_TYPE_BUTTON:
387 
388  if (!vcap.set_control(qctrl.id, 1))
389  std::cout << "Can not update control [" << qctrl.name << "] with value " << 1 << std::endl;
390 
391  break;
392 
393  case V4L2_CTRL_TYPE_STRING:
394  break;
395 
396  case V4L2_CTRL_TYPE_MENU:
397  case V4L2_CTRL_TYPE_INTEGER_MENU:
398 
399  Gtk::TreeModel::Children::iterator iter = static_cast<Gtk::ComboBox *>(wctrl)->get_active();
400  Gtk::TreeModel::Row row = *iter;
401  struct v4l2_querymenu qmenu;
402  qmenu = row[model_control_menu.m_col_data];
403 
404  if (!vcap.set_control(qctrl.id, qmenu.index))
405  std::cout << "Can not update control [" << qctrl.name << "] with value " << qmenu.name << std::endl;
406 
407  break;
408  }
409  __update_control_widgets(ctrl_list_default);
410  }
