179 bool v4lcap::init_mmap(int nbuffers) {
180  // = Get current format ====================================================
181  if(!get_format(&format_src, V4L2_BUF_TYPE_VIDEO_CAPTURE)) return false;
182 
183  // = Request buffers by mmap ===============================================
184  if(!request_buffers_mmap(&reqbuf, nbuffers)) {
185  std::cout << "Can't request mmap buffers.\n";
186  return false;
187  }
188 
189  if(reqbuf.count < 2) {
190  // = Request 0 buffers to clean memory driver ==========================
191  request_buffers_mmap(&reqbuf, 0);
192  std::cout << "Insufficient buffer memory.\n";
193  return false;
194  }
195 
196  // = Alloc buffer holders ==================================================
197  buffers.qtd = reqbuf.count;
198  buffers.buffs = (struct buffer *) calloc(reqbuf.count, sizeof(struct buffer));
199 
200  if(!buffers.buffs) return false;
201 
202  // = Memory map: mmap ======================================================
203  for (unsigned int i = 0; i < reqbuf.count; ++i) {
204  struct v4l2_buffer buffer = { 0 };
205 
206  if (!query_buffer(&buffer, V4L2_BUF_TYPE_VIDEO_CAPTURE, V4L2_MEMORY_MMAP, i))
207  std::cout << "Can't request buffer " << i << std::endl;
208 
209  buffers.buffs[i].length = buffer.length; /* remember for munmap() */
210 
211  if(emulate_format) {
212  buffers.buffs[i].start = v4l2_mmap(NULL, //
213  buffer.length, //
214  PROT_READ | PROT_WRITE, /* recommended */
215  MAP_SHARED, /* recommended */
216  fd, //
217  buffer.m.offset);
218  } else {
219  buffers.buffs[i].start = mmap(NULL, //
220  buffer.length, //
221  PROT_READ | PROT_WRITE, /* recommended */
222  MAP_SHARED, /* recommended */
223  fd, //
224  buffer.m.offset);
225  }
226 
227  if(MAP_FAILED == buffers.buffs[i].start) {
228  /* If you do not exit here you should unmap() and free() the buffers mapped so far. */
229  perror("mmap");
230  exit(EXIT_FAILURE);
231  }

256 bool v4lcap::start_capturing(enum memory_type mem_type, int nbuffers) {
257  switch(mem_type) {
258  case MEM_MMAP:
259  if (!init_mmap(nbuffers)) return false;
260  break;
261  case MEM_USER:
262  case MEM_READ:
263  default:
264  return false;
265  }
266 
267  if(!init_mmap(nbuffers)) return false;
268 
269  // = Enqueue all buffers ===================================================
270  for(unsigned int i = 0; i < reqbuf.count; ++i) {
271  struct v4l2_buffer buffer = { 0 };
272  if(!enqueue_buff(&buffer, reqbuf.type, reqbuf.memory, i))
273  std::cout << "Can't enqueue buffer " << i << std::endl;
274 
276  if(!query_buffer(&buffer, reqbuf.type, reqbuf.memory, i))
277  std::cout << "Can't request buffer " << i << std::endl;
278 
279  if(buffer.flags & V4L2_BUF_FLAG_QUEUED) {
280  std::cout << "Buffer " << i << " queued!" << std::endl;
281  } else {
282  std::cout << "Buffer " << i << " NOT queued!" << std::endl;
283  }
284  }
285 
286  this->format_dest = this->format_src;
287  format_dest.fmt.pix.pixelformat = V4L2_PIX_FMT_RGB24;
288 
289  if(format_src.fmt.pix.pixelformat != V4L2_PIX_FMT_RGB24) {
290  needs_convert = true;
291  } else {
292  needs_convert = false;
293  }
294 
295  if(needs_convert) {
296  v4l2_format copy = format_src;
297  v4lconvert_try_format(v4lconv, &format_dest, &format_src);
298  // v4lconvert_try_format sometimes modifies the source format if it thinks
299  // that there is a better format available. Restore our selected source
300  // format since we do not want that happening.
301  format_src = copy;
302  }
303 
304  return streamon(V4L2_BUF_TYPE_VIDEO_CAPTURE);
305 }
