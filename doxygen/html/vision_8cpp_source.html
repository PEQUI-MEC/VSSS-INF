147 int Vision::inSphere(Robots::_Status * robot, std::vector<Tag> * tempTags, cv::Point secondary) {
148  // se esta secundária faz parte do robô
149  if(calcDistance(robot->position, secondary) <= ROBOT_RADIUS) {
150  if(calcDistance(tempTags->at(0).frontPoint, secondary) < calcDistance(tempTags->at(0).rearPoint, secondary)) {
151  tempTags->at(0).switchPoints();
152  // recalcula a orientação com os novos pontos (isso só é feito uma vez em cada robô, se necessário)
153  robot->orientation = atan2((tempTags->at(0).frontPoint.y-robot->position.y)*1.3/height,(tempTags->at(0).frontPoint.x-robot->position.x)*1.5/width);
154  }
155 
156  double secSide = atan2((secondary.y-robot->position.y)*1.3/height,(secondary.x-robot->position.x)*1.5/width);
157 
158  // Cálculo do ângulo de orientação para diferenciar robôs de mesma cor
159  return (atan2(sin(secSide-robot->orientation+3.1415), cos(secSide-robot->orientation+3.1415))) > 0 ? 1 : -1;
160  }
161 
162  return 0;
163 }
