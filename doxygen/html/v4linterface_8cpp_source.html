347  void V4LInterface::__update_cb_frame_interval() {

350 
351  struct v4l2_format fmt;
352  unsigned int & width = fmt.fmt.pix.width;
353  unsigned int & height = fmt.fmt.pix.height;
354 
355  vcap.get_format(&fmt, V4L2_BUF_TYPE_VIDEO_CAPTURE);
356  sp_width.set_value((double) width);
357  sp_height.set_value((double) height);
358 
359  struct v4l2_frmivalenum frminterval;
360 
361  struct v4l2_streamparm streamparm;
362  vcap.get_stream_parameter(&streamparm, V4L2_BUF_TYPE_VIDEO_CAPTURE);
363 
364  if(vcap.enum_frame_interval(&frminterval, fmt.fmt.pix.pixelformat, width, height, 0, true)) {
365  do {
366  Gtk::TreeModel::Row row = *(ls_frame_interval->append());
367  float fps = (float) frminterval.discrete.denominator / (float) frminterval.discrete.numerator;
368  Glib::ustring name = Glib::ustring::compose("%1/%2 = %3 fps", frminterval.discrete.numerator, frminterval.discrete.denominator, fps);
369  row[model_frame_interval.m_col_name] = name;
370  row[model_frame_interval.m_col_data] = frminterval;
371 
372  if (streamparm.parm.capture.timeperframe.numerator == frminterval.discrete.numerator
373  && streamparm.parm.capture.timeperframe.denominator == frminterval.discrete.denominator)
374  {
375  cb_frame_interval.set_active(row);
376  }
377 
378  } while (vcap.enum_frame_interval(&frminterval, fmt.fmt.pix.pixelformat, width, height));
379  }
380  cb_frame_interval_signal.block(false);
